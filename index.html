Language Identification Using Machine Learning

language_identification.py 

# Step 1: Load the Dataset
with open("dataset.txt", "r", encoding="utf-8") as file:
    dataset = file.readlines()

# Step 2: Data Preprocessing
import re

def preprocess_text(text):
    # Convert text to lowercase
    text = text.lower()
    # Remove punctuation
    text = re.sub(r'[^\w\s]', '', text)
    # Tokenize the text
    tokens = text.split()
    return tokens

# Apply preprocessing to each text sample in the dataset
preprocessed_data = []
for line in dataset:
    # Split each line based on the "|" character
    parts = line.split("|")
    if len(parts) == 2:
        text = parts[0].strip()
        lang = parts[1].strip()
        preprocessed_data.append((preprocess_text(text), lang))

# Step 3: Feature Extraction using TF-IDF
from sklearn.feature_extraction.text import TfidfVectorizer

# Convert tokenized words back into single strings
preprocessed_texts_joined = [' '.join(tokens) for tokens, _ in preprocessed_data]

# Initialize the TF-IDF vectorizer
tfidf_vectorizer = TfidfVectorizer()

# Fit the vectorizer and transform data into TF-IDF vectors
tfidf_matrix = tfidf_vectorizer.fit_transform(preprocessed_texts_joined)

# Step 4: Splitting the Data and Model Training
from sklearn.model_selection import train_test_split
from sklearn.linear_model import LogisticRegression

# Split data into training and testing sets
X_train, X_test, y_train, y_test = train_test_split(
    tfidf_matrix, 
    [lang for _, lang in preprocessed_data], 
    test_size=0.2, 
    random_state=42
)

# Initialize and train a classifier
classifier = LogisticRegression()
classifier.fit(X_train, y_train)

# Step 5: Model Evaluation
from sklearn.metrics import classification_report

# Evaluate the classifier
y_pred = classifier.predict(X_test)
print(classification_report(y_test, y_pred))

# Step 6: Language Identification using Trained Model
def identify_language(input_text):
    # Preprocess the input text
    preprocessed_input = preprocess_text(input_text)
    
    # Convert to TF-IDF vectors
    input_vector = tfidf_vectorizer.transform([' '.join(preprocessed_input)])
    
    # Predict language
    predicted_language = classifier.predict(input_vector)[0]
    
    # Verify prediction
    dataset_languages = set(lang for _, lang in preprocessed_data)
    if predicted_language in dataset_languages:
        return predicted_language
    return "Language not recognized"

# Example usage
input_text = "c√≥mo te llamas"
result = identify_language(input_text)
print(f"Predicted language: {result}")

Rainwater Collection Estimation Using Computer Vision

rainwater_estimation.py 

import cv2
import numpy as np
import matplotlib.pyplot as plt
import os

def process_roof_image(image_path):
    """
    Process roof image to estimate collection area and potential rainwater harvesting.
    """
    # Read and validate image
    img = cv2.imread(image_path)
    if img is None:
        raise ValueError("‚ùå Error: Image could not be loaded.")
    
    # Convert to RGB for display
    img_rgb = cv2.cvtColor(img, cv2.COLOR_BGR2RGB)
    
    # Image preprocessing
    gray = cv2.cvtColor(img, cv2.COLOR_BGR2GRAY)
    blur = cv2.GaussianBlur(gray, (5, 5), 0)
    edges = cv2.Canny(blur, 50, 150)
    
    # Find and process contours
    contours, _ = cv2.findContours(edges, cv2.RETR_EXTERNAL, cv2.CHAIN_APPROX_SIMPLE)
    largest_contour = max(contours, key=cv2.contourArea)
    
    # Create mask and calculate area
    mask = np.zeros_like(gray)
    cv2.drawContours(mask, [largest_contour], -1, 255, -1)
    
    # Calculate area
    area_pixels = cv2.contourArea(largest_contour)
    scaling_factor = 0.25  # 1 pixel = 0.25 m¬≤
    area_m2 = area_pixels * scaling_factor
    
    return img, largest_contour, area_m2

def calculate_water_collection(area_m2, rainfall_mm, runoff_coefficient=0.9):
    """
    Calculate potential water collection based on roof area and rainfall.
    """
    water_collected_liters = area_m2 * rainfall_mm * runoff_coefficient
    water_collected_m3 = water_collected_liters / 1000
    return water_collected_liters, water_collected_m3

def visualize_results(img, largest_contour, area_m2, rainfall_scenarios=[10, 25, 50, 75, 100]):
    """
    Create visualizations of the results.
    """
    # Create overlay visualization
    img_overlay = img.copy()
    overlay = img.copy()
    cv2.drawContours(overlay, [largest_contour], -1, (0, 255, 0), -1)
    alpha = 0.4
    img_overlay = cv2.addWeighted(overlay, alpha, img_overlay, 1 - alpha, 0)
    
    # Calculate water collection for different scenarios
    runoff_coefficient = 0.9
    collected_water = [area_m2 * r * runoff_coefficient for r in rainfall_scenarios]
    
    # Create plots
    fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(15, 6))
    
    # Plot 1: Roof Detection
    ax1.imshow(cv2.cvtColor(img_overlay, cv2.COLOR_BGR2RGB))
    ax1.set_title("Detected Roof Area")
    ax1.axis("off")
    
    # Plot 2: Water Collection Estimates
    ax2.bar(rainfall_scenarios, collected_water, color='skyblue')
    ax2.set_xlabel("Rainfall (mm)")
    ax2.set_ylabel("Water Collected (liters)")
    ax2.set_title("Estimated Water Collection vs Rainfall")
    ax2.grid(True)
    
    plt.tight_layout()
    return fig

# Example usage
if __name__ == "__main__":
    # Process image and calculate area
    img, contour, area = process_roof_image("roof_image.jpg")
    
    # Calculate water collection for 50mm rainfall
    water_liters, water_m3 = calculate_water_collection(area, 50)
    
    # Create visualizations
    fig = visualize_results(img, contour, area)
    
    # Display results
    print(f"‚úÖ Analysis Complete!")
    print(f"üìè Estimated Roof Area: {area:.2f} m¬≤")
    print(f"üíß Water Collection Potential:")
    print(f"   50mm rainfall: {water_liters:.2f} liters ({water_m3:.2f} m¬≥)")

Project Overview

This project uses computer vision techniques to estimate potential rainwater collection from rooftops using aerial imagery. It combines image processing, contour detection, and area calculation to provide accurate estimates of collection potential.

Technical Implementation

  • Image preprocessing with Gaussian blur and Canny edge detection
  • Contour detection and analysis for roof area identification
  • Pixel-to-meter conversion for accurate area calculation
  • Interactive visualization with Matplotlib

Key Features

  • Automated roof area detection
  • Multiple rainfall scenario analysis
  • Visual overlay of detected areas
  • Water collection estimation in both liters and cubic meters
